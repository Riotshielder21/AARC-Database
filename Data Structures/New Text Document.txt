CREATE TABLE userTa(
  UserID Integer AUTO_INCREMENT NOT NULL,
  MemberName VARCHAR(30) NOT NULL,
  Username VARCHAR(30) NOT NULL,
  Password VARCHAR(30) DEFAULT 'password1' NOT NULL,
  Privilege TINYINT DEFAULT '0',
  PRIMARY KEY(UserID)
);
CREATE TABLE checkerTa(
  CheckID VARCHAR(15) NOT NULL,
  UserID INTEGER NOT NULL,
  PRIMARY KEY(CheckID),
  FOREIGN KEY(UserID) REFERENCES UserTa(UserID)
);
CREATE TABLE resourceTa(
  TypeID VARCHAR(15) NOT NULL,
  RName VARCHAR(30) NOT NULL,
  RLocation VARCHAR(255) NOT NULL,
  Filesize INTEGER,
  Filetype VARCHAR(6),
  FileAuthor VARCHAR(30),
  PRIMARY KEY(TypeID)
);
CREATE TABLE inventoryTa(
  ID INTEGER AUTO_INCREMENT NOT NULL,
  Qty INTEGER DEFAULT '1',
  DateChecked TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  CheckID VARCHAR(15) NOT NULL,
  TypeID VARCHAR(15) NOT NULL,
  PRIMARY KEY(ID),
  FOREIGN KEY(CheckID) REFERENCES CheckerTa(CheckID),
  FOREIGN KEY(TypeID) REFERENCES ResourceTa(TypeID)
);
CREATE TABLE PCNET (
	PCID INTEGER NOT NULL AUTO_INCREMENT,
    PCNAME VARCHAR(20) NOT NULL,
    PRIMARY KEY(PCID)
);
CREATE TABLE PCLISTING(
    LISTID INTEGER NOT NULL AUTO_INCREMENT,
    NAME VARCHAR(20) NOT NULL,
    PCTYPE INTEGER NOT NULL,
    PRIMARY KEY(LISTID),
    FOREIGN KEY (PCTYPE) REFERENCES PCNET(PCID)
);
